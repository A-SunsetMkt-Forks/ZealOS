//ZealOS supports standard Print()
//codes and extended ones.
//See Print("") Format Strings.

//Demonstrates aux_format_num codes.

"%10h?d\n", 123456789;
"%10h?d\n", 12345;

"%10h3d\n", 123456789;
"%10h3d\n", 12345;

"\nWith the ',' flag:\n";
"%10,h3d\n", 123456789;
"%10,h3d\n", 12345;

"\nFloating Point:\n";
"%16h?n\n", .00123456789;
"%16h?n\n", 12345.0;

"%16.2h-6n\n", .00123456789;
"%16.2h-6n\n", 12345.0;

'\n';

"%h25c\n", 'x'; //Print 25 x's

"%h*c\n", 25, 'y'; //Print 25 y's

'zzzz\n'; //PutChars() up to 8 characters.

'\n';
