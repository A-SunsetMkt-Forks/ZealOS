U8 i;

"Ed(\"InEd.DD\");\n";
//Go to start of file
Message(MESSAGE_KEY_DOWN, 0, SC_CURSOR_UP | SCF_CTRL);
for (i = 0; i <= 0xC3; i++)
{
        if (i)
                Message(MESSAGE_KEY_DOWN_UP, 0, SC_F3);
        else
                "%c0x%c", CH_CTRLF, CH_ESC;
        Message(MESSAGE_KEY_DOWN, 0, SC_CURSOR_RIGHT);
        Message(MESSAGE_KEY_DOWN, 0, SC_CURSOR_RIGHT);
        "%02X", i;
        Message(MESSAGE_KEY_DOWN, 0, SC_CURSOR_RIGHT);
        Message(MESSAGE_KEY_DOWN, 0, SC_CURSOR_RIGHT);
        "%c%c", CH_BACKSPACE, CH_BACKSPACE;
}
